            Converting ES5 codes to ES6

            -------------------------------
            REFACTOR ES5 CODE TO ES6 (Enhanced Object Literals)
            ES5 code
            var a = 1,
              b = 2,
              c = 3;
            obj = {
              a: a,
              b: b,
              c: c,
            };

            obj.a = 1, obj.b = 2, obj.c = 3

            -------------------------------

            -------------------------------
            ES5 code
            var lib = {
              sum: function (a, b) {
                return a + b;
              },
              mult: function (a, b) {
                return a * b;
              },
            };

            console.log(lib.sum(2, 3)); // 5
            console.log(lib.mult(2, 3)); // 6
